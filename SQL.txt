CREATE VIEW [dbo].[Address]
AS
SELECT        dbo.Provincia.Descripcion AS 'Provincia', dbo.Canton.Descripcion AS 'Canton', dbo.Distrito.Descripcion AS 'Distrito', dbo.Distrito.Senales
FROM            dbo.Provincia INNER JOIN
                         dbo.Canton ON dbo.Provincia.Id_Provincia = dbo.Canton.Id_Provincia INNER JOIN
                         dbo.Distrito ON dbo.Canton.Id_Canton = dbo.Distrito.Id_Canton
